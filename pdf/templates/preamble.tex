    \def\bcbcalloutbg{BCBImportantBg}%
    \def\bcbcallouttitlecolor{BCBImportantTitle}%
  }{}%
  \ifstrequal{#1}{math}{%
    \def\bcbcalloutbg{BCBMathBg}%
    \def\bcbcallouttitlecolor{BCBMathTitle}%
  }{}%
  \ifstrequal{#1}{action}{%
    \def\bcbcalloutbg{BCBActionBg}%
    \def\bcbcallouttitlecolor{BCBActionTitle}%
  }{}%
  \ifstrequal{#1}{assessment}{%
    \def\bcbcalloutbg{BCBAssessmentBg}%
    \def\bcbcallouttitlecolor{BCBAssessmentTitle}%
  }{}%
  \ifstrequal{#1}{warning}{%
    \def\bcbcalloutbg{BCBWarningBg}%
    \def\bcbcallouttitlecolor{BCBWarningTitle}%
  }{}%
  \ifstrequal{#1}{tip}{%
    \def\bcbcalloutbg{BCBTipBg}%
    \def\bcbcallouttitlecolor{BCBTipTitle}%
  }{}%
  \ifstrequal{#1}{caution}{%
    \def\bcbcalloutbg{BCBCautionBg}%
    \def\bcbcallouttitlecolor{BCBCautionTitle}%
  }{}%
}
